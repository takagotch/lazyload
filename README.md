```js
LazyLoad.js('http://example.com/foo.js', function(){
  alert('foo.js has been loaded');
});

LazyLoad.js(['foo.js', 'bar.js', 'baz.js'], function(){
  alert('all files have been loaded');
});

LazyLoad.js(['foo.js', 'bar.js', 'baz.js'], function(){
  alert(arg);
}, 'foo.css has been loaded');

LazyLoad.css('foo.css', function(){
  alert(this.foo);
}, null, {foo: 'bar'});
```
